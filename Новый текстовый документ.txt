// Сохраняем текущее значение скролла
var scrollPosition = window.pageYOffset;

// Запрещаем скролл у body
document.body.style.overflow = 'hidden';

// Устанавливаем значение скролла обратно на то, что было до открытия меню
document.body.style.top = -scrollPosition + 'px';
document.body.style.position = 'fixed';

function closeMenu() {
  
  document.body.style.overflow = '';

 
  document.body.style.top = '';
  document.body.style.position = '';

  window.scrollTo(0, scrollPosition);
}


function openMenu() {

  document.body.style.overflow = 'hidden';

  
  document.body.style.top = -scrollPosition + 'px';
  document.body.style.position = 'fixed';

 
  document.querySelector('.dropdown-menu').classList.add('active');
  document.querySelector('.page-header__burger').classList.add('active');


  document.querySelector('.dropdown-menu__close-icon').addEventListener('click', closeMenu);
}

document.querySelector('.page-header__burger').addEventListener('click', openMenu);
